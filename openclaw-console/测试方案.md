## 4. Mock API 方案

### 4.1 MSW (Mock Service Worker) 方案

推荐使用 MSW 拦截 fetch 请求，模拟真实的 HTTP 响应。

**安装依赖**:
```bash
npm install -D msw @mswjs/msw
```

**创建 handler (src/api/mocks/handlers.ts)**:
```typescript
import { http, HttpResponse } from 'msw'
import { API_BASE_URL } from '../gateway'

export const handlers = [
  // 网关状态
  http.get(`${API_BASE_URL}/status`, () => {
    return HttpResponse.json({
      online: true,
      uptime: '3天 12时 34分',
      cpu: 23,
      memory: 512,
      messages_today: 128,
      api_calls: 2048,
    })
  }),

  // 任务列表
  http.get(`${API_BASE_URL}/tasks`, () => {
    return HttpResponse.json({
      tasks: [
        { id: '1', name: '早间汇报', description: '每日汇报', status: 'active', nextRun: '明天 08:00', schedule: '0 8 * * *', lastRun: '今天 08:00', lastStatus: 'success' },
      ]
    })
  }),

  // 切换任务状态
  http.post(`${API_BASE_URL}/tasks/:id/toggle`, ({ params }) => {
    return HttpResponse.json({
      id: params.id,
      name: '早间汇报',
      status: 'inactive',
    })
  }),
]
```

**初始化 MSW (src/test-setup.ts)**:
```typescript
import { setupServer } from 'msw/node'
import { handlers } from './api/mocks/handlers'

export const server = setupServer(...handlers)

beforeAll(() => server.listen())
afterEach(() => server.resetHandlers())
afterAll(() => server.close())
```

## 5. 测试示例

### 5.1 Sidebar 组件测试

```typescript
import { render, screen, fireEvent } from '@testing-library/react'
import { Sidebar } from '../components/Sidebar'
import { useAppStore } from '../store'

// 重置 store 状态
beforeEach(() => {
  useAppStore.setState({
    currentPage: 'dashboard',
    sidebarCollapsed: false,
  })
})

describe('Sidebar', () => {
  it('渲染所有菜单项', () => {
    render(<Sidebar />)
    expect(screen.getByText('仪表盘')).toBeInTheDocument()
    expect(screen.getByText('对话')).toBeInTheDocument()
    expect(screen.getByText('任务管理')).toBeInTheDocument()
  })

  it('点击菜单切换页面', () => {
    render(<Sidebar />)
    
    fireEvent.click(screen.getByText('任务管理'))
    
    expect(useAppStore.getState().currentPage).toBe('tasks')
  })

  it('折叠/展开侧边栏', () => {
    render(<Sidebar />)
    
    const toggleButton = screen.getByRole('button', { name: /收起/i })
    fireEvent.click(toggleButton)
    
    expect(useAppStore.getState().sidebarCollapsed).toBe(true)
  })
})
```

### 5.2 Tasks 页面测试

```typescript
import { render, screen, fireEvent } from '@testing-library/react'
import { Tasks } from '../pages/Tasks'
import { useAppStore } from '../store'

// Mock react-hot-toast
jest.mock('react-hot-toast', () => ({
  toast: {
    success: jest.fn(),
    error: jest.fn(),
  },
  Toaster: () => null,
}))

describe('Tasks', () => {
  beforeEach(() => {
    useAppStore.setState({ tasks: [
      { id: '1', name: '测试任务', description: '测试', status: 'active', nextRun: '12:00', schedule: '0 12 * * *' }
    ]})
  })

  it('渲染任务列表', () => {
    render(<Tasks />)
    expect(screen.getByText('测试任务')).toBeInTheDocument()
  })

  it('添加新任务', () => {
    render(<Tasks />)
    
    fireEvent.click(screen.getByText('添加任务'))
    fireEvent.change(screen.getByPlaceholderText('例如: 早间汇报'), { 
      target: { value: '新任务' } 
    })
    fireEvent.change(screen.getByPlaceholderText('例如: 0 8 * * *'), { 
      target: { value: '0 9 * * *' } 
    })
    fireEvent.click(screen.getByText('保存'))
    
    const tasks = useAppStore.getState().tasks
    expect(tasks).toHaveLength(2)
    expect(tasks[1].name).toBe('新任务')
  })

  it('删除任务', () => {
    // Mock confirm
    window.confirm = jest.fn(() => true)
    
    render(<Tasks />)
    
    const deleteButton = screen.getByText('删除')
    fireEvent.click(deleteButton)
    
    expect(useAppStore.getState().tasks).toHaveLength(0)
  })
})
```

### 5.3 Store 测试

```typescript
import { useAppStore } from '../store'

describe('useAppStore', () => {
  beforeEach(() => {
    useAppStore.setState(useAppStore.getState())
  })

  it('toggleSidebar 切换侧边栏状态', () => {
    const { toggleSidebar } = useAppStore.getState()
    
    expect(useAppStore.getState().sidebarCollapsed).toBe(false)
    
    toggleSidebar()
    expect(useAppStore.getState().sidebarCollapsed).toBe(true)
    
    toggleSidebar()
    expect(useAppStore.getState().sidebarCollapsed).toBe(false)
  })

  it('setCurrentPage 设置当前页面', () => {
    const { setCurrentPage } = useAppStore.getState()
    
    setCurrentPage('settings')
    expect(useAppStore.getState().currentPage).toBe('settings')
  })

  it('addTask 添加任务', () => {
    const { addTask, tasks } = useAppStore.getState()
    const initialCount = tasks.length
    
    addTask({
      id: '99',
      name: '新任务',
      description: '描述',
      status: 'active',
      nextRun: '12:00',
      schedule: '0 12 * * *',
    })
    
    expect(useAppStore.getState().tasks).toHaveLength(initialCount + 1)
  })
})
```
